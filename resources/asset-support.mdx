export const chainLabel = (raw) => CHAIN_NAMES[raw] ?? raw;

export const AssetTable = () => {
  const [tokens, setTokens] = React.useState([]);
  const [loading, setLoading] = React.useState(true);
  const [error, setError] = React.useState(null);
  const [search, setSearch] = React.useState("");
  const [sortBy, setSortBy] = React.useState("symbol");

  React.useEffect(() => {
    fetch("https://1click.chaindefuser.com/v0/tokens")
      .then((r) => r.json())
      .then((data) => { setTokens(data); setLoading(false); })
      .catch(() => { setError("Failed to load token data."); setLoading(false); });
  }, []);

  const filtered = tokens
    .filter((t) => {
      const q = search.toLowerCase();
      return (
        t.symbol.toLowerCase().includes(q) ||
        chainLabel(t.blockchain).toLowerCase().includes(q)
      );
    })
    .sort((a, b) => {
      const valA = sortBy === "blockchain" ? chainLabel(a.blockchain) : a.symbol;
      const valB = sortBy === "blockchain" ? chainLabel(b.blockchain) : b.symbol;
      return valA.localeCompare(valB);
    });

  const BORDER = "1px solid rgba(255,255,255,0.08)";
  const CELL = { padding: "12px 16px" };
  const CELL_FIRST = { padding: "12px 16px 12px 0" };
  const CELL_LAST = { padding: "12px 0 12px 16px" };
  const ORANGE = "#f97316";

  return (
    <div>
      <div style={{ display: "flex", gap: "12px", marginBottom: "20px", flexWrap: "wrap", alignItems: "center" }}>
        <input
          type="text"
          placeholder="Search by symbol or blockchain…"
          value={search}
          onChange={(e) => setSearch(e.target.value)}
          style={{
            padding: "8px 12px",
            borderRadius: "6px",
            border: "1px solid rgba(255,255,255,0.15)",
            fontSize: "14px",
            flex: "1",
            minWidth: "200px",
            background: "#000000",
            color: "#ffffff",
            outline: "none",
          }}
        />

        {/* Toggle switch */}
        <div style={{ display: "flex", alignItems: "center", gap: "10px" }}>
          <span style={{ fontSize: "14px", color: "#9ca3af", whiteSpace: "nowrap" }}>Sort by:</span>
          <div style={{
            display: "flex",
            background: "rgba(255,255,255,0.06)",
            borderRadius: "8px",
            padding: "3px",
            gap: "2px",
          }}>
            {["symbol", "blockchain"].map((opt) => (
              <button
                key={opt}
                onClick={() => setSortBy(opt)}
                style={{
                  padding: "5px 14px",
                  borderRadius: "6px",
                  border: "none",
                  cursor: "pointer",
                  fontSize: "13px",
                  fontWeight: sortBy === opt ? "600" : "400",
                  background: sortBy === opt ? ORANGE : "transparent",
                  color: sortBy === opt ? "#ffffff" : "#9ca3af",
                  transition: "all 0.15s ease",
                  textTransform: "capitalize",
                }}
              >
                {opt === "symbol" ? "Symbol" : "Blockchain"}
              </button>
            ))}
          </div>
        </div>
      </div>

      {loading && <p>Loading tokens…</p>}
      {error && <p style={{ color: "red" }}>{error}</p>}

      {!loading && !error && (
        <>
          <p style={{ fontSize: "13px", color: "#6b7280", marginBottom: "12px" }}>
            {filtered.length} token{filtered.length !== 1 ? "s" : ""} found
          </p>
          <div style={{ width: "100%", fontSize: "14px" }}>
            <div style={{ display: "flex", borderBottom: BORDER }}>
              <div style={{ ...CELL_FIRST, flex: 1, fontWeight: "600" }}>Symbol</div>
              <div style={{ ...CELL_LAST, flex: 1, fontWeight: "600" }}>Blockchain</div>
            </div>
            {filtered.map((t) => (
              <div key={t.assetId} style={{ display: "flex", borderBottom: BORDER }}>
                <div style={{ ...CELL_FIRST, flex: 1, fontWeight: "500" }}>{t.symbol}</div>
                <div style={{ ...CELL_LAST, flex: 1 }}>{chainLabel(t.blockchain)}</div>
              </div>
            ))}
          </div>
        </>
      )}
    </div>
  );
};
